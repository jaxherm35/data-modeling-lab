users can create recipes with ingredients and instructions
recipes can be marked as public or private
users can view other people’s recipes
ingredients from recipes can be added to user’s grocery lists
users can create their own occasions and assign recipes to occasions

Columns:


Users
        UserID - integer
        Password- varchar
        Email- varchar
        Username- varchar
        Recipes- text
CREATE TABLE users (
  userID SERIAL PRIMARY KEY,
  password VARCHAR(25),
  email VARCHAR(50),
  username VARCHAR(25),
  recipes TEXT
  );




Shopping List
        Ingredients- text
        Shopping list- text


CREATE TABLE list (
  ingredients TEXT,
  list TEXT
  );












Recipes
        Ingredients- varchar
        Instructions- varchar
        User id- integer
        Occasion- varchar
CREATE TABLE recipes (
  ingredients TEXT,
  instructions TEXT,
  userID INT REFERENCES users(userID),
  occasion VARCHAR(50)
  );


Groups
        userId- integer
        Group Id-
